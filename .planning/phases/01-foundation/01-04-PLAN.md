    import { db } from "@/lib/db";
    import { topics, lessons } from "@/lib/db/schema";
    import { eq, asc } from "drizzle-orm";

    export async function getTopics() {
      return db.query.topics.findMany({
        orderBy: [asc(topics.order)],
      });
    }

    export async function getTopicWithLessons(slug: string) {
      return db.query.topics.findFirst({
        where: eq(topics.slug, slug),
        with: {
          lessons: {
            orderBy: [asc(lessons.order)],
          },
        },
      });
    }

    4. Create `src/app/topics/[topic]/page.tsx` as a server component:
    ```tsx
    import { requireAuth } from "@/actions/auth";
    import { getTopicWithLessons } from "@/lib/content/queries";
    import { notFound } from "next/navigation";
    import { TopicTabs } from "@/components/topics/TopicTabs";
    import { Badge } from "@/components/ui/badge";

    export default async function TopicPage({
      params,
    }: {
      params: Promise<{ topic: string }>;
    }) {
      const { topic: slug } = await params;
      await requireAuth();

      const topic = await getTopicWithLessons(slug);
      if (!topic) notFound();

      return (
        <div className="px-6 lg:px-12 py-8">
          <div className="mb-6">
            <Badge variant="secondary" className="mb-2">{topic.title}</Badge>
            <h1 className="text-3xl font-semibold">{topic.title}</h1>
            <p className="text-muted-foreground mt-1">{topic.description}</p>
          </div>
          <TopicTabs topic={topic} lessons={topic.lessons} />
        </div>
      );
    }
    ```
    NOTE: Next.js 16 uses `params: Promise<{ topic: string }>` -- must await params. Call `requireAuth()` before data fetch per INFRA-04. Valid slugs: "huffman", "n-ary-trees", "red-black-trees", "b-trees" per D-09. Invalid slugs return 404 via `notFound()`.
