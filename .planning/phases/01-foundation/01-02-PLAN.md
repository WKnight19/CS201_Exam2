    **Step 2: Create Better Auth server instance**

    Create `src/lib/auth.ts`:
    ```typescript
    import { betterAuth } from "better-auth";
    import { drizzleAdapter } from "better-auth/adapters/drizzle";
    import { db } from "@/lib/db";

    export const auth = betterAuth({
      baseURL: process.env.BETTER_AUTH_URL,
      secret: process.env.BETTER_AUTH_SECRET,
      database: drizzleAdapter(db, {
        provider: "pg",
      }),
      socialProviders: {

          clientId: process.env.GOOGLE_CLIENT_ID as string,
          clientSecret: process.env.GOOGLE_CLIENT_SECRET as string,
        },
      },
    });
    ```
    IMPORTANT: Include `secret` field with `process.env.BETTER_AUTH_SECRET`. Include `baseURL` with `process.env.BETTER_AUTH_URL`. Both are required for production. The `drizzleAdapter` import is from `better-auth/adapters/drizzle` (NOT a separate package).

    **Step 5: Create middleware for optimistic cookie-based route protection**

    Create `middleware.ts` at project root (NOT in src/):
    ```typescript
    import { NextRequest, NextResponse } from "next/server";
    import { getSessionCookie } from "better-auth/cookies";

    export function middleware(request: NextRequest) {
      const sessionCookie = getSessionCookie(request);

      if (!sessionCookie) {
        return NextResponse.redirect(new URL("/sign-in", request.url));
      }

      return NextResponse.next();
    }

    export const config = {
      matcher: ["/", "/topics/:path*"],
    };
    ```
    The matcher protects `/` (dashboard) and `/topics/*` (topic pages). The `/sign-in` route is NOT protected. The `/api/auth/*` routes are NOT protected (auth callbacks need to be public).

    **Step 6: Create requireAuth server-side helper**

    Create `src/actions/auth.ts`:
    ```typescript
    "use server";

    import { auth } from "@/lib/auth";
    import { headers } from "next/headers";
    import { redirect } from "next/navigation";

    export async function requireAuth() {
      const session = await auth.api.getSession({
        headers: await headers(),
      });
      if (!session) redirect("/sign-in");
      return session;
    }
    ```
    This performs a full DB-validated session check per INFRA-04. Middleware is UX-only; this is the security guard. Every Server Component and Server Action that touches user data must call this.

    **Step 7: Create sign-in page**

    Create `src/app/(auth)/sign-in/page.tsx` as a client component:
    ```tsx
    "use client";

    import { useState } from "react";
    import { authClient } from "@/lib/auth-client";
    import { Button } from "@/components/ui/button";
    import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";

    export default function SignInPage() {
      const [isLoading, setIsLoading] = useState(false);
      const [error, setError] = useState<string | null>(null);

      const handleSignIn = async () => {
        setIsLoading(true);
        setError(null);
        try {
          await authClient.signIn.social({ provider: "google" });
        } catch {
          setError("Sign-in failed. Try again or check your Google account settings.");
          setIsLoading(false);
        }
      };

      return (
        <div className="flex min-h-screen items-center justify-center px-4">
          <Card className="w-full max-w-[400px]">
            <CardHeader className="text-center">
              <CardTitle className="text-2xl font-semibold">CS 201 Study App</CardTitle>
              <p className="text-sm text-muted-foreground">Sign in to study</p>
            </CardHeader>
            <CardContent>
              <Button
                onClick={handleSignIn}
                disabled={isLoading}
                className="w-full"
              >
                {isLoading ? "Signing in..." : "Sign in with Google"}
              </Button>
              {error && (
                <p className="mt-4 text-center text-sm text-destructive">{error}</p>
              )}
            </CardContent>
          </Card>
        </div>
      );
    }
    ```
    Per UI-SPEC: centered card ~400px max-width, accent fill button, loading state changes text to "Signing in...", error message below button. Copy text exactly: "Sign in to study" subheading, "Sign in with Google" button, "Sign-in failed. Try again or check your Google account settings." error.
