    1. Create `drizzle.config.ts` at project root:
    ```typescript
    import { defineConfig } from "drizzle-kit";

    export default defineConfig({
      schema: "./src/lib/db/schema.ts",
      out: "./drizzle",
      dialect: "postgresql",
      dbCredentials: {
        url: process.env.DATABASE_URL!,
      },
    });
    ```

    2. Create `src/lib/db/index.ts` -- Drizzle instance with neon-http driver:
    ```typescript
    import { drizzle } from "drizzle-orm/neon-http";
    import { neon } from "@neondatabase/serverless";
    import * as schema from "./schema";

    const sql = neon(process.env.DATABASE_URL!);
    export const db = drizzle(sql, { schema });
    ```
    IMPORTANT: Use `drizzle-orm/neon-http` (NOT `drizzle-orm/neon-serverless`). Use `neon()` function (NOT `new Pool()`). This is the stateless HTTP driver pattern required for Vercel serverless.

    3. Create `src/lib/db/schema.ts` with ALL tables per D-04. The schema must include:
    - `topics` table: id (uuid PK), slug (text, unique), title (text), description (text), order (integer), createdAt (timestamp with tz)
    - `lessons` table: id (uuid PK), topicId (uuid FK -> topics.id), title (text), conceptMd (text), pseudocodeMd (text), cppCode (text), order (integer), createdAt (timestamp with tz)
    - `questionTypeEnum` pgEnum: "multiple_choice", "fill_blank", "trace_step", "debug_step", "short_answer"
    - `questions` table: id (uuid PK), topicId (uuid FK -> topics.id), questionType (enum), prompt (text), content (jsonb), explanation (text nullable), difficulty (integer default 2), createdAt
    - `flashcards` table: id (uuid PK), topicId (uuid FK -> topics.id), front (text), back (text), createdAt
    - `userTopicCards` table: id (uuid PK), userId (TEXT not uuid -- Better Auth user.id is text), topicId (uuid FK), FSRS fields (stability text default "0", difficulty text default "0", elapsedDays int default 0, scheduledDays int default 0, reps int default 0, lapses int default 0, state int default 0, due timestamp), createdAt, updatedAt, unique constraint on (userId, topicId)
    - `userQuestionCards` table: same pattern as userTopicCards but with questionId FK -> questions.id, unique on (userId, questionId)
    - `reviewHistory` table: id (uuid PK), userId (text), questionId (uuid FK), rating (int), correct (boolean), answeredAt (timestamp), index on (userId, answeredAt)

    Use exact column definitions from RESEARCH.md Pattern 6. All `userId` columns must be `text("user_id")` NOT `uuid("user_id")` because Better Auth generates text IDs.

    Do NOT include Better Auth tables (user, session, account, verification) in this file yet -- those will be generated by the Better Auth CLI in Plan 02 and merged into this file.

    4. Add Drizzle relations for query builder (`.query.topics.findFirst({ with: { lessons: true } })`):
    ```typescript
    import { relations } from "drizzle-orm";

    export const topicsRelations = relations(topics, ({ many }) => ({
      lessons: many(lessons),
      questions: many(questions),
      flashcards: many(flashcards),
    }));

    export const lessonsRelations = relations(lessons, ({ one }) => ({
      topic: one(topics, { fields: [lessons.topicId], references: [topics.id] }),
    }));

    export const questionsRelations = relations(questions, ({ one }) => ({
      topic: one(topics, { fields: [questions.topicId], references: [topics.id] }),
    }));

    export const flashcardsRelations = relations(flashcards, ({ one }) => ({
      topic: one(topics, { fields: [flashcards.topicId], references: [topics.id] }),
    }));
    ```

    5. Create `src/lib/db/question-schemas.ts` -- Zod discriminated union per D-05:
    ```typescript
    import { z } from "zod/v4";

    const multipleChoiceContent = z.object({
      type: z.literal("multiple_choice"),
      options: z.array(z.string()).min(2).max(6),
      correctIndex: z.number().int().min(0),
    });

    const fillBlankContent = z.object({
      type: z.literal("fill_blank"),
      blanks: z.array(z.object({ id: z.string(), answer: z.string() })),
    });

    const traceStepContent = z.object({
      type: z.literal("trace_step"),
      steps: z.array(z.unknown()),
      correctStepIndex: z.number().int(),
    });

    const debugStepContent = z.object({
      type: z.literal("debug_step"),
      steps: z.array(z.unknown()),
      errorStepIndex: z.number().int(),
    });

    const shortAnswerContent = z.object({
      type: z.literal("short_answer"),
      rubric: z.array(z.string()),
      sampleAnswer: z.string(),
    });

    export const questionContentSchema = z.discriminatedUnion("type", [
      multipleChoiceContent,
      fillBlankContent,
      traceStepContent,
      debugStepContent,
      shortAnswerContent,
    ]);

    export type QuestionContent = z.infer<typeof questionContentSchema>;
    ```
    NOTE: Zod 4 uses `import { z } from "zod/v4"` not `import { z } from "zod"`. Verify the import path works with the installed zod@4.3.6.
